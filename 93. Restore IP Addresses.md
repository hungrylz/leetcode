# [Problem](https://leetcode.com/problems/restore-ip-addresses/)

#### DFS
````
    public List<String> restoreIpAddresses(String s) {
        List<String> res = new ArrayList<>();
        helper(res,s,new ArrayList<>(),1,0);
        return res;
    }
    private void helper(List<String> res, String s, List<String> cur, int part, int start){
        if(part == 4){
            if(s.length()-start > 3) return;
            String left = s.substring(start);
            if(left.length() >= 2 && left.charAt(0) == '0') return;
            int l = Integer.parseInt(left);
            if(l > 255) return;
            cur.add(left);
            printf(res,cur);
            cur.remove(cur.size()-1);
            return;
        }
        for(int i = 1; i < 4; i++){
            if(start + i >= s.length()) break;
            String d = s.substring(start,start+i);
            if(d.length() >= 2 && d.charAt(0) == '0') break;
            if(i == 3){
                int l = Integer.parseInt(d);
                if(l > 255) continue;
            }
            // if(d.equals("00"))break;
            cur.add(d);
            helper(res,s,cur,part+1,start+i);
            cur.remove(cur.size()-1);
        }
        
    }
    private void printf(List<String> res, List<String>cur){
        StringBuilder sb = new StringBuilder(cur.get(0));
        int index = 1;
        for(int i = 0; i < 3; i++){
            sb.append(".");
            sb.append(cur.get(index++));
        }
        res.add(sb.toString());
    }
````
