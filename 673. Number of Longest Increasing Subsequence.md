# [Problem](https://leetcode.com/problems/number-of-longest-increasing-subsequence/)


#### DP
````
class Solution {
    public int findNumberOfLIS(int[] nums) {
        int n = nums.length, ans = Integer.MIN_VALUE;
        int[] dp = new int[n];
        int[] count = new int[n];
        for(int i = 0; i < n; i++){
            dp[i] = 1;
            count[i] = 1;
            for(int j = 0; j < i; j++){
                if(nums[j] < nums[i]){
                    if(dp[j] + 1 == dp[i]){
                        count[i] += count[j];
                    }else if(dp[j] + 1 > dp[i]){
                        count[i] = count[j];
                        dp[i] = dp[j] + 1;
                    }
                }
            }
            ans = Math.max(dp[i],ans);
        }
        int res = 0;
        for(int i = 0; i < n; i++){
            if(dp[i] == ans){
                res += count[i];
            }
        }
        return res;
    }
}
````
